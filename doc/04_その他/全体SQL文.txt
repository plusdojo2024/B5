//ユーザーテーブル//
CREATE TABLE users (
  id INT AUTO_INCREMENT,
  user_id VARCHAR(100) NOT NULL,
  password VARCHAR(32) NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, 
PRIMARY KEY (ID));

//対戦履歴//
CREATE TABLE historys (
  id INT AUTO_INCREMENT,
  user_id INT,
  match_day DATE DEFAULT CURRENT_TIMESTAMP,
  rank INT,
  point INT,
  people INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));

//対戦相手//
CREATE TABLE opponents (
  id INT AUTO_INCREMENT,
  user_id INT,
  history_id INT,
  name VARCHAR(20),
  rank INT,
  point INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));

//履歴詳細//
CREATE TABLE history_details (
  id INT AUTO_INCREMENT,
  user_id INT,
  history_id INT,
  wind INT,
  game INT,
  number INT,
  seat_wind INT,
  point INT,
  han INT,
  hu INT,
  result INT,
  title INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));

//手牌//
CREATE TABLE hands (
  id INT AUTO_INCREMENT,
  user_id INT,
  history_id INT,
  tile INT,
  game_num INT,
  dispose_num INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));

//ドラ表示牌//
CREATE TABLE doras (
  id INT AUTO_INCREMENT,
  user_id INT,
  history_id INT,
  reverse INT,
  tile INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));

//手役//
CREATE TABLE teyakus (
  id INT AUTO_INCREMENT,
  user_id INT,
  history_id INT,
  teyaku INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));

//フレンドデータ//
CREATE TABLE friends (
  id INT AUTO_INCREMENT,
  user_id INT NOT NULL,
  friend_name VARCHAR(100) NOT NULL,
  word VARCHAR(100),
  like_yaku VARCHAR(100),
  record VARCHAR(30),
  count_yaku VARCHAR(1000),
  image INT,
  last_login TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));


//マイページ//
CREATE TABLE mypages (
  id INT AUTO_INCREMENT,
  user_id VARCHAR NOT NULL,
  name VARCHAR(100),
  word VARCHAR(100),
  like_yaku VARCHAR(30),
  record VARCHAR(30),
  count_yaku VARCHAR(1000),
  image INT,
  last_login TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));

//user_idのユニーク化//
CREATE UNIQUE INDEX user_idINDEX
  ON MYPAGES(USER_ID);

//牌一覧//
CREATE TABLE tiles (
  id INT AUTO_INCREMENT,
  tile_name VARCHAR(20) NOT NULL,
  tile_kind INT NOT NULL,
  tile_number INT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));

//役一覧//
CREATE TABLE yakus (
  id INT AUTO_INCREMENT,
  yaku_name VARCHAR(20) NOT NULL,
  han INT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (id));
